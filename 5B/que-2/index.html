Problem Statement 2:

Q. Write a code in any programming language that counts the digits in a number.
                        
#include <bits/stdc++.h>
using namespace std;
     
int countDigit(n)
{
    int count = 0;
    while (n != 0)
    {
        n = n / 10;
        ++count;
    }
return count;
}
     
int main(void)
{
    int n;
    cout << "Enter Number: ";
    cin >> n;
    cout << "Number of digits: " << countDigit(n);
    return 0;
}
                        
                    
                    

input > Enter Number: 1234

output > Number of digits: 4
